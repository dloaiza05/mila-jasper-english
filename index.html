📖 Instructivo y ejemplo del código

Pídele a Claude, Gemini o ChatGPT las preguntas con este formato y pégalas en la ventana de la derecha. Hay 9 tipos de pregunta interactivos, incluida 📚 lectura de textos.

{
  "titulo": "Mi quiz de inglés",
  "preguntas": [

    { "formato":"opcion",
      "q":"The dog ___ busy.",
      "opciones":[
        {"texto":"is","correcta":true,"razon":"He/She/It usa is."},
        {"texto":"are","correcta":false,"razon":"are es plural."},
        {"texto":"am","correcta":false,"razon":"am solo con I."}
      ],
      "frase":"The dog is busy.",
      "traduccion":"El perro está ocupado.",
      "pronunciacion":"de dog is bísi",
      "regla":"He/She/It -> is", "dato":"To be = ser y estar.",
      "glosario":[
        {"palabra":"dog","es":"perro",
         "tipo":"sustantivo","pron":"dog"},
        {"palabra":"busy","es":"ocupado",
         "tipo":"adjetivo","pron":"bísi"}
      ] },

    { "formato":"multiple",
      "q":"Marca TODOS los pronombres:",
      "opciones":[
        {"texto":"he","correcta":true,"razon":"él"},
        {"texto":"they","correcta":true,"razon":"ellos"},
        {"texto":"table","correcta":false,"razon":"es sustantivo"}
      ] },

    { "formato":"vf",
      "q":"'I are happy' es correcto.",
      "respuesta":false, "razon":"Con I se usa am.",
      "frase":"I am happy.", "traduccion":"Yo estoy feliz.",
      "pronunciacion":"ái am jápi" },

    { "formato":"escribir",
      "q":"Escribe el número 40 en inglés.",
      "respuesta":"forty", "traduccion":"cuarenta",
      "pronunciacion":"fórti" },

    { "formato":"completar",
      "q":"Completa con el verbo to be:",
      "frase":"She ___ a teacher and they ___ students.",
      "respuestas":["is","are"],
      "traduccion":"Ella es profesora y ellos son estudiantes." },

    { "formato":"huecos",
      "q":"Completa tocando las palabras:",
      "frase":"The ___ is ___.",
      "respuestas":["dog","busy"],
      "banco":["dog","cat","busy","happy"],
      "traduccion":"El perro está ocupado." },

    { "formato":"ordenar",
      "q":"Ordena la frase:",
      "palabras":["I","am","a","carpenter"],
      "traduccion":"Yo soy carpintero." },

    { "formato":"emparejar",
      "q":"Une cada palabra con su significado:",
      "pares":[
        {"en":"dog","es":"perro"},
        {"en":"wood","es":"madera"},
        {"en":"table","es":"mesa"}
      ] },

    { "formato":"lectura",
      "q":"Lee el texto: My Day",
      "texto":"I am a carpenter. I wake up at six...\n\nI work with wood every day...",
      "traduccion":"Soy carpintero. Me despierto a las seis...\n\nTrabajo con madera todos los días...",
      "glosario":[{"palabra":"carpenter","es":"carpintero","tipo":"sustantivo","pron":"cárpenter"}],
      "frases":[{"frase":"wake up","es":"despertarse","pron":"uéik ap"}],
      "preguntas":[
        { "formato":"opcion","q":"What time does he wake up?",
          "opciones":[
            {"texto":"At six","correcta":true,"razon":"El texto lo dice."},
            {"texto":"At nine","correcta":false,"razon":"Relee el texto."}
          ] }
      ] }
  ]
}
